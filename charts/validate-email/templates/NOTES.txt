validate-email has been deployed.

1. Expose the service locally:

  kubectl --namespace {{ .Release.Namespace }} port-forward service/{{ include "validate-email.fullname" . }} 8080:{{ .Values.service.port }}

2. Get your API key:

  {{- if .Values.existingSecret }}
  # Using existing secret: {{ .Values.existingSecret }}
  API_KEY=$(kubectl -n {{ .Release.Namespace }} get secret {{ .Values.existingSecret }} -o jsonpath="{.data.api_keys}" | base64 -d)
  {{- else if .Values.apiKeys }}
  # API key was provided via values
  API_KEY="{{ .Values.apiKeys }}"
  {{- else }}
  # An API key was auto-generated. Retrieve it with:
  API_KEY=$(kubectl -n {{ .Release.Namespace }} get secret {{ include "validate-email.fullname" . }} -o jsonpath="{.data.api_keys}" | base64 -d)
  {{- end }}

3. Validate an email:

  curl -s http://127.0.0.1:8080/validate_email \
    -X POST \
    -H "Content-Type: application/json" \
    -H "X-API-Key: $API_KEY" \
    -d '{"email":"user@example.com"}'

4. Health check:

  curl http://127.0.0.1:8080/healthz

5. Uninstall:

  helm uninstall {{ .Release.Name }} --namespace {{ .Release.Namespace }}
